Object Detection‚Äã

Detect objects using ML models:

import { useFrameProcessor, detectObjects } from 'react-native-video-pro';

function VideoWithDetection() {
  const [detections, setDetections] = useState([]);

  const frameProcessor = useFrameProcessor((frame) => {
    'worklet';
    
    const results = detectObjects(frame, {
      model: 'yolov5',
      confidence: 0.5,
    });
    
    // Send results back to JS thread
    runOnJS(setDetections)(results);
    
    return frame;
  }, []);

  return (
    <View>
      <Video
        source={{ uri: 'https://example.com/video.mp4' }}
        frameProcessor={frameProcessor}
        style={styles.video}
      />
      <DetectionOverlay detections={detections} />
    </View>
  );
}

Face Detection‚Äã

const frameProcessor = useFrameProcessor((frame) => {

    contourMode: 'none',
  });
  
  faces.forEach(face => {
    console.log('Face bounds:', face.bounds);
    console.log('Smile probability:', face.smilingProbability);
  });
  
  return frame;
}, []);

Performance Tips‚Äã

  1. Use worklets - Always include 'worklet' directive
  2. Minimize JS bridge calls - Batch updates with runOnJS
  3. Optimize processing frequency - Skip frames if needed
  4. Use GPU when possible - Leverage hardware acceleration
const frameProcessor = useFrameProcessor((frame) => {
  'worklet';
  
  // Skip every other frame for performance
  if (frame.timestamp % 2 === 0) {
    return frame;

}, []);
